SECTION 3 — DATABASE MIGRATIONS AUDIT
Check the following tables and columns exist in the database:

-- Check jobs table columns
SELECT column_name FROM information_schema.columns 
WHERE table_name = 'jobs' 
AND column_name IN (
  'upsell_shown', 'upsell_accepted', 'upsell_declined', 
  'upsell_deferred', 'upsell_cadence_selected',
  'property_vacant', 'first_recurring_discounted',
  'last_cleaned_response', 'last_cleaned_flag',
  'overage_disclaimer_acknowledged', 'overage_rate',
  'applied_bundle_id', 'bundle_discount_total'
);

-- Check rate_locks table exists
SELECT EXISTS (
  SELECT FROM information_schema.tables 
  WHERE table_name = 'rate_locks'
);

-- Check offer_settings table exists  
SELECT EXISTS (
  SELECT FROM information_schema.tables 
  WHERE table_name = 'offer_settings'
);

-- Check addon_bundles table exists
SELECT EXISTS (
  SELECT FROM information_schema.tables 
  WHERE table_name = 'addon_bundles'
);

-- Check Phes has offer_settings seeded
SELECT * FROM offer_settings 
WHERE company_id = (SELECT id FROM companies WHERE slug = 'phes-cleaning');

-- Check Appliance Bundle seeded
SELECT b.name, b.discount_value, b.active, 
       array_agg(i.addon_key) as items
FROM addon_bundles b
JOIN addon_bundle_items i ON i.bundle_id = b.id
WHERE b.company_id = (SELECT id FROM companies WHERE slug = 'phes-cleaning')
GROUP BY b.id, b.name, b.discount_value, b.active;

-- Check recurring scope has hourly rates set
SELECT name, weekly_hourly_rate, biweekly_hourly_rate, 
       monthly_hourly_rate, use_frequency_hourly_rates
FROM pricing_scopes
WHERE name ILIKE '%recurring%'
AND company_id = (SELECT id FROM companies WHERE slug = 'phes-cleaning');

-- Check move_date column was dropped
SELECT column_name FROM information_schema.columns 
WHERE table_name = 'jobs' AND column_name = 'move_date';
-- Should return 0 rows


Report each check as EXISTS/MISSING or correct value/incorrect value.
