NEW DB TABLES TO CREATE
═══════════════════════════════════════════════════════════

loyalty_tiers (
  id SERIAL PRIMARY KEY,
  company_id UUID REFERENCES companies(id),
  tier_name TEXT NOT NULL,
  min_visits INTEGER DEFAULT 0,
  min_lifetime_revenue NUMERIC DEFAULT 0,
  reward_description TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
)

client_loyalty (
  id SERIAL PRIMARY KEY,
  client_id UUID REFERENCES clients(id),
  company_id UUID REFERENCES companies(id),
  tier_id INTEGER REFERENCES loyalty_tiers(id),
  tier_override TEXT,
  points_balance INTEGER DEFAULT 0,
  total_points_earned INTEGER DEFAULT 0,
  notes TEXT,
  updated_at TIMESTAMPTZ DEFAULT NOW()
)

referrals (
  id SERIAL PRIMARY KEY,
  company_id UUID REFERENCES companies(id),
  referrer_client_id UUID REFERENCES clients(id),
  referred_name TEXT NOT NULL,
  referred_phone TEXT,
  referred_email TEXT,
  status TEXT DEFAULT 'pending',
  reward_issued BOOLEAN DEFAULT FALSE,
  reward_amount NUMERIC,
  source TEXT DEFAULT 'manual',
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
)

All three tables require RLS policies scoped to company_id.
Supabase RLS pattern: company_id = (SELECT company_id FROM users 
WHERE id = auth.uid())
